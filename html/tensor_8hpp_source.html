227template <typename T>
228inline std::pair<unsigned int, unsigned int> neo::tensor<T>::loc_interval(int* args, unsigned int cap) const{
229 unsigned int size_copy{_size};
230
231 unsigned int running_index{0};
232 for(std::size_t i{}; i < cap; i++){
233 size_copy/=_shape[i];
234 running_index += (args[i]*size_copy);
235 }
236
237 std::pair<unsigned int, unsigned int> range(running_index, running_index+size_copy-1);
238 return range;
239}

277template <typename... Ta>
278inline neo::tensor<T>& neo::tensor<T>::get(Ta... args) const{
279 constexpr unsigned int n = sizeof...(Ta);
280 assert_check(n >= 1,"Must pass at least one argument");
281 assert_check(n <= _rank, "Indexing deeper than number of dimensions");
282 int arr[n]{args...};
283
284 // Check that arguments are within bounds
285 for (std::size_t i{}; i < n; i++){
286 assert_check(arr[i]<_shape[i], "Index out of bounds for dimension");
287 }
288
289 std::pair<unsigned int, unsigned int> range = loc_interval(arr, n);
290 neo::tensor<T>* new_tensor = new neo::tensor<T>(_array+range.first, _shape+n, _rank-n, false);
291 return *new_tensor;
292}
